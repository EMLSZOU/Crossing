# 定义

Python 可使用语法糖`[]`表示列表，其中的元素可以是任何类型，以顺序存储的方式动态储存数据。

# 常见操作

下面是一些列表的常见操作：

```python
In [1]: a = [1, 2, 3]

In [2]: b = [4, 5, 6]

In [3]: a + b
Out[3]: [1, 2, 3, 4, 5, 6]

In [4]: a * 2
Out[4]: [1, 2, 3, 1, 2, 3]

In [5]: a[-1]
Out[5]: 3

In [6]: a[0:2]
Out[6]: [1, 2]

In [7]: a[::-1]
Out[7]: [3, 2, 1]

In [8]: a[::2]
Out[8]: [1, 3]

In [9]: for key, value in enumerate(a):
    ...:     print('{} --> {}'.format(key, value))
    ...:     
0 --> 1
1 --> 2
2 --> 3

In [10]: len(a)
Out[10]: 3

In [11]: max(a)
Out[11]: 3

In [12]: min(a)
Out[12]: 1

In [13]: reversed(a)
Out[13]: <list_reverseiterator at 0x7fabec65be80>

In [14]: [i for i in reversed(a)]
Out[14]: [3, 2, 1]

In [15]: sorted(a, key=abs, reverse=True)
Out[15]: [3, 2, 1]

In [16]: sum(a)
Out[16]: 6

In [17]: for x, y in zip(a, b):
    ...:     print(x, y)
    ...:     
1 4
2 5
3 6

In [18]: [i for i in dir(list) if not i.startswith('__')]
Out[18]: 
['append',
 'clear',
 'copy',
 'count',
 'extend',
 'index',
 'insert',
 'pop',
 'remove',
 'reverse',
 'sort']

In [19]: L = [[1, 2, 3], [4, 5, 6], [7, 8, 9]]

In [20]: r = [sum(i) for i in L]    # 列表生成式

In [21]: r
Out[21]: [6, 15, 24]

In [22]: [a for a in range(0, 10) if a % 2 == 0]
Out[22]: [0, 2, 4, 6, 8]
```
